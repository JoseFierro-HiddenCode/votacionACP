rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ✅ NUEVO: Permitir configuración (Lectura pública, Escritura autenticada)
    // Esto es necesario para que el admin.js pueda abrir/cerrar votación
    match /config/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // votos: solo crear una vez
    match /votes/{userId} {
      allow create: if request.auth != null;
      allow read: if request.auth != null;
      allow update, delete: if false;
    }

    // participantes: leer libremente, escribir solo logueados
    match /participants/{teamId} {
      allow read: if true;
      allow create, update: if request.auth != null;
      allow delete: if false;
    }
  }
}
